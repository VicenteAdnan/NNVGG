# NeuralNetworks VGG
First install next libraries in Linux(Ubuntu)
Tensorflow:
  pip3 install tensorflow
Keras:
  pip3 install keras
OpenCV:
     sudo apt update
     sudo apt-get install libopencv-dev
     sudo apt install python3-opencv 
Scikit Learn:
     pip3 install -U scikit-learn
Matplotlib:
    python3 -m pip install -U matplotlib 
Searborn:
    pip install seaborn
Numpy:
    pip3 install numpy
